------------------------------------------------
* dorađivanje koda
  * natklasa      →  štampaj, zameni, napravi
  * cpp/hpp       →  ne sme da ima samo hpp
  * višeprolazno  →  zamena continue i slično
------------------------------------------------
* while na for →  bez izmena

while (uslov) {
  telo;
}

for (; uslov;) {
  telo;
}
------------------------------------------------
* do na for → uslovna promenljiva

do {
  telo;
} while (uslov);

{
  int cond = 1;
  for (; cond; cond = uslov) {
    telo;
  }
}
------------------------------------------------
* for na while →  zamena continue

for (init; uslov; inc) {
  telo;
}

{
  init;
  while (uslov) {
    telo[continue; → {
                        inc;
                        continue;
                     }];
    inc;
  }
}
------------------------------------------------
* do na while → preko for (ponavljanje)
------------------------------------------------
* for na do → zamena continue (ponavljanje)
------------------------------------------------
